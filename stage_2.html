Ticket 2: Problème de synchronisation lié au site

Il y avait un problème de synchronisation entre les différentes langues du site que j'ai donc du modifier. Où le pays d'origine n'était pas pris en compte et par défaut c'était donc la France. 


//AddressType.php
use Symfony\Component\Form\Extension\Core\Type\TextareaType;
use Symfony\Component\Form\Extension\Core\Type\TextType;
use Symfony\Component\Form\FormBuilderInterface;
use Symfony\Component\Form\FormEvent;
use Symfony\Component\Form\FormEvents;
use Symfony\Component\OptionsResolver\OptionsResolver;
use Symfony\Contracts\Translation\TranslatorInterface;

public function buildForm(FormBuilderInterface $builder, array $options): void
    {
        $defaultCountry = $this->countryRepository->findOneBy(['code' => 'FR']);

        $builder->addEventListener(FormEvents::PRE_SET_DATA, function (FormEvent $event) use ($defaultCountry): void {
            $address = $event->getData();
            if (null === $address) {
                $address = new Address();
                $address->setCountry($defaultCountry);
                $event->setData($address);
            } elseif ($address instanceof Address && null === $address->getCountry()) {
                $address->setCountry($defaultCountry);
            }
        });

        $builder
            ->add('company', TextType::class, [
                'label' => $this->translator->trans('order.detail.address.company'),
            ])
        ;
    }
